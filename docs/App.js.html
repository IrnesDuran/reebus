import React, {useState, useEffect} from 'react';
import './App.css';

/**
 * Sample component
 *
 * @component
 * @example
 * return (
 *     <App />
 * )
 */
function App() {

    const API = process.env.REACT_APP_API;
    const API_URI = `${API}/users`;

    const [users, setUsers] = useState([]);

    useEffect(() => {
        fetch(API_URI)
            .then(response => response.json())
            .then(data => {
                setUsers(data);
            });
    });


    return (
     <div className="App">
        <header className="App-header">
        <h1>Welcome to WalterCode Starter</h1>
        <ul className="nobull">
            {
                users.map(user => (
                    <li key={user.id}>{user.username} => {user.email}</li>
                ))
            }
        </ul>
        </header>
    </div>
  );
}

export default App;
